const express = require('express'); const http = require('http'); const socketIO = require('socket.io'); const mongoose = require('mongoose'); const bodyParser = require('body-parser'); const app = express(); const server = http.createServer(app); const io = socketIO(server); // MongoDB Setup mongoose.connect('mongodb://localhost:27017/mememe', { useNewUrlParser: true, useUnifiedTopology: true }); const User = mongoose.model('User', new mongoose.Schema({ username: String, password: String })); const Meme = mongoose.model('Meme', new mongoose.Schema({ image: String, topText: String, bottomText: String, comments: [{ username: String, text: String }] })); // Middleware app.use(bodyParser.json()); app.use(express.static(__dirname)); // Routes app.post('/signup', async (req, res) => { const { username, password } = req.body; const exists = await User.findOne({ username }); if (exists) return res.status(400).send('User exists'); await User.create({ username, password }); res.send('Signup successful'); }); app.post('/login', async (req, res) => { const { username, password } = req.body; const user = await User.findOne({ username, password }); if (!user) return res.status(401).send('Invalid credentials'); res.send('Login successful'); }); app.get('/memes', async (req, res) => { const memes = await Meme.find(); res.json(memes); }); app.post('/memes', async (req, res) => { const meme = await Meme.create(req.body); io.emit('newMeme', meme); res.send('Meme posted'); }); app.post('/comment', async (req, res) => { const { memeId, username, text } = req.body; const meme = await Meme.findById(memeId); meme.comments.push({ username, text }); await meme.save(); io.emit('newComment', { memeId, username, text }); res.send('Comment added'); }); // Serve HTML app.get('/', (req, res) => { res.sendFile(__dirname + '/index.html'); }); // Socket.IO io.on('connection', socket => { console.log('User connected'); }); server.listen(3000, () => console.log('Server running on http://localhost:3000'));
